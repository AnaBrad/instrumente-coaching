# WOOP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `woop.html` — a single-file WOOP coaching exercise (Wish-Outcome-Obstacle-Plan, after Oettingen), deployed to GitHub Pages alongside the other Roti-Coaching tools. Supports save-mid-exercise, a history of completed WOOPs, and PDF export.

**Architecture:** Single self-contained HTML file with embedded CSS and vanilla JS. Reuses the existing design system (paper palette, Fraunces+Inter fonts) and patterns from sibling files: localStorage resume from `roata-increderii.html`, lazy-loaded PDF export from `roata-emotiilor.html`. The mockup `mockup-woop.html` already contains the final visual structure for all 7 screens — Task 1 starts from there.

**Tech Stack:** HTML5, vanilla CSS (CSS variables, grid, flexbox), vanilla JS, Google Fonts (Fraunces + Inter), localStorage API, html2canvas 1.4.1 (lazy-loaded), jspdf 2.5.1 (lazy-loaded).

**Spec:** `docs/superpowers/specs/2026-05-25-woop-design.md`

**No test framework.** Verification is visual/functional in the browser. Each task includes specific browser checks.

## File Structure

- **Create:** `woop.html` — the WOOP app (~1400 lines, all 7 screens in one file with show/hide)
- **Modify:** `index.html` — add a new card linking to `woop.html`
- **Reference (read-only):** `mockup-woop.html` (visual reference), `modelul-grow.html`, `roata-emotiilor.html` (PDF lazy-load pattern), `roata-increderii.html` (localStorage pattern)

```html
    localStorage.removeItem(DRAFT_KEY);

  const raw = localStorage.getItem(DRAFT_KEY);
  currentDraft = raw ? JSON.parse(raw) : null;
}

function saveHistory() {
  localStorage.setItem(HISTORY_KEY, JSON.stringify(history));
}

  const raw = localStorage.getItem(HISTORY_KEY);
  history = raw ? JSON.parse(raw) : [];
}

function savePrefs() {
  localStorage.setItem(PREF_KEY, JSON.stringify(prefs));
}

function loadPrefs() {
  const raw = localStorage.getItem(PREF_KEY);
  if (raw) prefs = Object.assign(prefs, JSON.parse(raw));
}

// Load everything on page boot
loadDraft();
loadHistory();
loadPrefs();
```

```js
// ─── LANDING ───────────────────────────────────────────
function buildLanding() {
  // Resume banner shown only if a draft exists
  const banner = document.getElementById('resume-banner');
  banner.hidden = !currentDraft;

  // History list
  const container = document.getElementById('history-container');
  if (history.length === 0) {
    container.innerHTML = `
      <div class="history-empty">Aici vor apărea WOOP-urile tale completate.</div>
    `;
    return;
  }
  const ul = document.createElement('ul');
  ul.className = 'history-list';
  history.forEach(w => {
    const li = document.createElement('li');
    li.className = 'history-item';
    li.innerHTML = `
      <div>
        <div class="history-wish">${escapeHtml(w.wish)}</div>
        <div class="history-meta">${formatDate(w.completedAt)} · ${TIMEFRAMES[w.timeframe].label.toLowerCase()}</div>
      </div>
      <div class="history-actions">
        <button type="button" class="icon-btn" data-action="view"   data-id="${w.id}">Vezi</button>
        <button type="button" class="icon-btn" data-action="pdf"    data-id="${w.id}">PDF</button>
        <button type="button" class="icon-btn" data-action="delete" data-id="${w.id}">Șterge</button>
      </div>
    `;
    ul.appendChild(li);
  });
  container.innerHTML = '';
  container.appendChild(ul);

  // Wire history row actions via delegation
  ul.addEventListener('click', e => {
    const btn = e.target.closest('button.icon-btn');
    if (!btn) return;
    const id = btn.dataset.id;
    const woop = history.find(w => w.id === id);
    if (!woop) return;
    if (btn.dataset.action === 'view') {
      viewingHistoryId = id;
      goTo('result');
    } else if (btn.dataset.action === 'pdf') {
      exportPDF(woop);  // wired in Task 12; defined as no-op stub for now
    } else if (btn.dataset.action === 'delete') {
      if (confirm(`Sigur ștergi WOOP-ul „${woop.wish}"?`)) {
        history = history.filter(w => w.id !== id);
        saveHistory();
        buildLanding();
      }
    }
  });
}

// Helpers
function escapeHtml(s) {
  return String(s ?? '').replace(/[&<>"']/g, c => ({
    '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'
  }[c]));
}

function formatDate(iso) {
  if (!iso) return '';
  const d = new Date(iso);
  const months = ['ian.', 'feb.', 'mar.', 'apr.', 'mai', 'iun.', 'iul.', 'aug.', 'sept.', 'oct.', 'nov.', 'dec.'];
  return `${d.getDate()} ${months[d.getMonth()]} ${d.getFullYear()}`;
}
```

```js
// ─── TIMEFRAME ─────────────────────────────────────────
function buildTimeframe() {
  const grid = document.getElementById('timeframe-grid');
  grid.innerHTML = '';
  Object.entries(TIMEFRAMES).forEach(([key, tf]) => {
    const div = document.createElement('div');
    div.className = 'timeframe-chip' + (currentDraft?.timeframe === key ? ' selected' : '');
    div.dataset.key = key;
    div.setAttribute('role', 'button');
    div.setAttribute('tabindex', '0');
    div.innerHTML = `
      <div class="tf-label">${tf.label}</div>
      <div class="tf-hint">${tf.hint}</div>
    `;
    div.addEventListener('click', () => selectTimeframe(key));
    div.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); selectTimeframe(key); }
    });
    grid.appendChild(div);
  });
  updateTimeframeContinue();
}

function selectTimeframe(key) {
  if (!currentDraft) return;
  currentDraft.timeframe = key;
  saveDraft();
  buildTimeframe();
}

function updateTimeframeContinue() {
  const btn = document.getElementById('tf-continue');
  btn.disabled = !currentDraft?.timeframe;
  btn.classList.toggle('disabled', !currentDraft?.timeframe);
}
```

```js
// ─── IMAGERY TIMER ─────────────────────────────────────
// Reusable countdown for Outcome and Obstacle screens.
// Usage: imageryTimer({ displayEl, startBtn, skipBtn, textarea, onComplete })
// - startBtn click → starts countdown, disables startBtn, updates displayEl
// - skipBtn click  → marks skipImageryByDefault=true and immediately completes
// - On completion: enables textarea, calls onComplete()
// - If prefs.skipImageryByDefault is true at start, auto-completes after a 200ms beat
// - First time ever (prefs.imageryCompletedOnce=false), skipBtn is hidden
function imageryTimer({ displayEl, startBtn, skipBtn, textarea, onComplete }) {
  const DURATION = 60; // seconds
  let remaining = DURATION;
  let intervalId = null;

  function render() {
    const m = String(Math.floor(remaining / 60)).padStart(2, '0');
    const s = String(remaining % 60).padStart(2, '0');
    displayEl.textContent = `${m}:${s}`;
  }

  function complete() {
    if (intervalId) clearInterval(intervalId);
    intervalId = null;
    displayEl.textContent = 'Gata.';
    startBtn.disabled = true;
    startBtn.classList.add('disabled');
    startBtn.textContent = 'Imaginare completă';
    textarea.disabled = false;
    textarea.classList.remove('locked');
    if (!prefs.imageryCompletedOnce) {
      prefs.imageryCompletedOnce = true;
      savePrefs();
    }
    setTimeout(() => textarea.focus(), 50);
    if (onComplete) onComplete();
  }

  // Initial UI state
  textarea.disabled = true;
  textarea.classList.add('locked');
  startBtn.disabled = false;
  startBtn.classList.remove('disabled');
  startBtn.textContent = 'Începe';
  remaining = DURATION;
  render();

  // Skip button visibility
  if (skipBtn) {
    skipBtn.hidden = !prefs.imageryCompletedOnce;
    skipBtn.onclick = () => {
      prefs.skipImageryByDefault = true;
      savePrefs();
      complete();
    };
  }

  // Start button
  startBtn.onclick = () => {
    if (intervalId) return;
    startBtn.disabled = true;
    startBtn.classList.add('disabled');
    intervalId = setInterval(() => {
      remaining -= 1;
      if (remaining <= 0) {
        complete();
      } else {
        render();
      }
    }, 1000);
  };

  // Auto-skip if user has set the preference
  if (prefs.skipImageryByDefault) {
    setTimeout(complete, 200);
  }
}
```

```js
function loadScript(src) {
  return new Promise((resolve, reject) => {
    const s = document.createElement('script');
    s.src = src;
    s.onload = resolve;
    s.onerror = () => reject(new Error('Failed to load ' + src));
    document.head.appendChild(s);
  });
}

async function ensurePdfLibs() {
  if (!window.html2canvas) {
    await loadScript('https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js');
  }
  if (!window.jspdf) {
    await loadScript('https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js');
  }
}

function slugify(text) {
  const diacritics = { 'ă':'a', 'â':'a', 'î':'i', 'ș':'s', 'ş':'s', 'ț':'t', 'ţ':'t', 'Ă':'a', 'Â':'a', 'Î':'i', 'Ș':'s', 'Ş':'s', 'Ț':'t', 'Ţ':'t' };
  return String(text || '')
    .toLowerCase()
    .replace(/[ăâîșşțţ]/gi, c => diacritics[c] || c)
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '')
    .slice(0, 40);
}

async function exportPDF(woop) {
  await ensurePdfLibs();

  // Build a temporary off-screen render card identical to the result layout
  const tf = TIMEFRAMES[woop.timeframe];
  const wrapper = document.createElement('div');
  wrapper.className = 'pdf-export-wrapper';
  wrapper.innerHTML = `
    <header class="top">
      <div>
        <h1>WOOP-ul tău</h1>
        <p class="subtitle">${formatDate(woop.completedAt)} · orizont: ${tf.label.toLowerCase()}</p>
      </div>
    </header>
    <div class="result-card">
      <div class="result-block">
        <div class="result-label">Wish — dorința</div>
        <div class="result-value wish">${escapeHtml(woop.wish)}</div>
      </div>
      <div class="result-block">
        <div class="result-label">Outcome — cel mai bun rezultat</div>
        <div class="result-value">${escapeHtml(woop.outcome)}</div>
      </div>
      <div class="result-block">
        <div class="result-label">Obstacle — obstacolul interior</div>
        <div class="result-value">${escapeHtml(woop.obstacle)}</div>
      </div>
      <div class="result-block">
        <div class="result-label">Plan — dacă-atunci</div>
        <div class="result-value"><em>Dacă</em> ${escapeHtml(woop.planIf)}, <em>atunci</em> ${escapeHtml(woop.planThen)}.</div>
      </div>
    </div>
  `;
  document.body.appendChild(wrapper);

  try {
    const canvas = await window.html2canvas(wrapper, {
      backgroundColor: '#f1e6d2',
      scale: 2,
    });
    const imgData = canvas.toDataURL('image/png');
    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF({ orientation: 'p', unit: 'mm', format: 'a4' });
    const pageWidth = 210;
    const pageHeight = 297;
    const imgWidth = pageWidth - 20; // 10mm margins
    const imgHeight = (canvas.height * imgWidth) / canvas.width;
    pdf.addImage(imgData, 'PNG', 10, 10, imgWidth, Math.min(imgHeight, pageHeight - 20));
    const filename = `woop-${slugify(woop.wish) || 'fara-titlu'}-${woop.completedAt.slice(0, 10)}.pdf`;
    pdf.save(filename);
  } finally {
    wrapper.remove();
  }
}
```
